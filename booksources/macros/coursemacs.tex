    {\def\optfile{#1}
      \ifx\optfile\empty\else\message{VLE EXERCISE #1}\index[programming]{#1}\fi
      \begin{frame}[containsverbatim]
        \ifx\optfile\empty
        \frametitle{Exercise \arabic{excounter}}

        \frametitle{Exercise \arabic{excounter} (\tt{#1})}
        \fi
        \refstepcounter{excounter}}
    {\end{frame}}
